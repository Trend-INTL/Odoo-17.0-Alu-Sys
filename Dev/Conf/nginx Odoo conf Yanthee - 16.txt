#odoo16 server
upstream odoo16 {
server 127.0.0.1:8069;
}
upstream odoochat {
server 127.0.0.1:8072;
}


server {
listen 80;
server_name odoo16.trend-ai.tech;

proxy_read_timeout 720s;
proxy_connect_timeout 720s;
proxy_send_timeout 720s;


# Add Headers for odoo16 proxy mode
proxy_set_header X-Forwarded-Host $host;
proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
proxy_set_header X-Forwarded-Proto $scheme;
proxy_set_header X-Real-IP $remote_addr;

# log
access_log /var/log/nginx/odoo16.access.log;
error_log /var/log/nginx/odoo16.error.log;

# Redirect requests to odoo16 backend server
location / {
proxy_redirect off;
proxy_pass http://odoo16;
}
location /longpolling {
proxy_pass http://odoo16chat;
}

# common gzip
gzip_types text/css text/less text/plain text/xml application/xml application/json application/javascript;
gzip on;


client_body_in_file_only clean;
client_body_buffer_size 32K;
client_max_body_size 500M;
sendfile on;
send_timeout 600s;
keepalive_timeout 300;
}
#####################################################



command
sudo ln -s /etc/nginx/sites-available/odoo16.conf /etc/nginx/sites-enabled/odoo16.conf
sudo nginx -t
sudo service nginx stop
sudo service nginx start
