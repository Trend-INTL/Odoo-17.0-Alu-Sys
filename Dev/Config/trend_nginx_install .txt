sudo nano /etc/nginx/sites-available/trend.conf
##################################################################
#odoo 17.0 server
upstream trend {
server 127.0.0.1:8075;
}
upstream trendchat {
server 127.0.0.1:8072;
}


server {
listen 80;
server_name server_name trend-ai.net www.trend-ai.net;

proxy_read_timeout 720s;
proxy_connect_timeout 720s;
proxy_send_timeout 720s;


# Add Headers for odoo proxy mode
proxy_set_header X-Forwarded-Host $host;
proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
proxy_set_header X-Forwarded-Proto $scheme;
proxy_set_header X-Real-IP $remote_addr;

# log
access_log /var/log/nginx/trend.access.log;
error_log /var/log/nginx/trend.error.log;

# Redirect requests to odoo backend server
location / {
proxy_redirect off;
proxy_pass http://trend;
}
location /longpolling {
proxy_pass http://trendchat;
}

# common gzip
gzip_types text/css text/less text/plain text/xml application/xml application/json application/javascript;
gzip on;


client_body_in_file_only clean;
client_body_buffer_size 32K;
client_max_body_size 500M;
sendfile on;
send_timeout 600s;
keepalive_timeout 300;
}
#####################################################

sudo ln -s /etc/nginx/sites-available/trend.conf /etc/nginx/sites-enabled/trend.conf
sudo nginx -t
sudo systemctl reload nginx
sudo service trend restart
sudo systemctl reload nginx
